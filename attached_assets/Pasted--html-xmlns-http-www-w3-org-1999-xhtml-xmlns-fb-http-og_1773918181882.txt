        function getTermo(idioma,id,descricao,caixa) {
            var msgAux = "";
            try {
                msgAux = eval("id_" + id + "_" + idioma);
				if (msgAux == "") {
					msgAux = descricao;
				}
            } catch(e) {
                msgAux = descricao;
            }
            if (caixa == 1) {
                msgAux = msgAux.toUpperCase();
            } else if (caixa == 2) {
                msgAux = msgAux.toLowerCase();
            }
			if ( msgAux.indexOf("'") != -1 ){
				msgAux = msgAux.replace("'", "\'");
			}else {
			    msgAux = msgAux.replace('\'', '\\\'');
            }
            return msgAux;
